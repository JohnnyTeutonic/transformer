181void Matrix::apply_gelu_derivative(const Matrix& x) {
182 constexpr float sqrt_2_over_pi = 0.7978845608028654f;
183 if (size() != x.size()) {
184 throw std::runtime_error("Matrix dimensions must match for GELU derivative");
185 }
186
187 if (data_.empty() || x.data_.empty()) {
188 throw std::runtime_error("Empty matrix in GELU derivative");
189 }
190
191#pragma omp parallel for simd
192 for (size_t i = 0; i < size(); i++) {
193 if (i >= x.data_.size() || i >= data_.size()) {
194 throw std::runtime_error("Index out of bounds in GELU derivative");
195 }
196
197 float val = x.data_[i];
198 val = std::clamp(val, -10.0f, 10.0f);
199
200 float cdf = 0.5f * (1.0f + std::tanh(sqrt_2_over_pi * (val + 0.044715f * val * val * val)));
201 float pdf = sqrt_2_over_pi * (1.0f + 3.0f * 0.044715f * val * val);
202 float derivative = cdf + val * pdf * (1.0f - std::tanh(val) * std::tanh(val));
203 derivative = std::clamp(derivative, -10.0f, 10.0f);
204 data_[i] *= derivative;
205 }
206}

208void Matrix::apply_softmax() {
209#pragma omp parallel for
210 for (size_t i = 0; i < rows_; ++i) {
211 float max_val = -std::numeric_limits<float>::infinity();
212#pragma omp simd reduction(max : max_val)
213 for (size_t j = 0; j < cols_; ++j) {
214 max_val = std::max(max_val, (*this)(i, j));
215 }
216
217 float sum = 0.0f;
218#pragma omp simd reduction(+ : sum)
219 for (size_t j = 0; j < cols_; ++j) {
220 float exp_val = std::exp((*this)(i, j) - max_val);
221 (*this)(i, j) = exp_val;
222 sum += exp_val;
223 }
224
225#pragma omp simd
226 for (size_t j = 0; j < cols_; ++j) {
227 (*this)(i, j) /= sum;
228 }
229 }
230}
