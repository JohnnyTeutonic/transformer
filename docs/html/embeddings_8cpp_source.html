29Matrix TokenEmbedding::forward(const std::vector<int>& tokens) {
30 // Input validation
31 if (tokens.empty()) {
32 throw std::runtime_error("Empty token sequence");
33 }
34 for (int token : tokens) {
35 if (token < 0 || static_cast<size_t>(token) >= vocab_size_) {
36 throw std::runtime_error("Token id " + std::to_string(token) + " out of range [0, " +
37 std::to_string(vocab_size_) + ")");
38 }
39 }
40
41 Matrix output(tokens.size(), embedding_dim_);
42
43 // Copy embeddings
44 for (size_t i = 0; i < tokens.size(); ++i) {
45 for (size_t j = 0; j < embedding_dim_; ++j) {
46 float val = weights_(tokens[i], j);
47 if (std::isnan(val) || std::isinf(val)) {
48 throw std::runtime_error("Invalid embedding value at position (" +
49 std::to_string(tokens[i]) + "," + std::to_string(j) +
50 "): " + std::to_string(val));
51 }
52 output(i, j) = val;
53 }
54 }
55
56 // Normalize output embeddings with better numerical stability
57 const float eps = 1e-6f; // Increased epsilon for stability
58 for (size_t i = 0; i < tokens.size(); i++) {
59 float row_norm = 0.0f;
60 // Compute norm for this embedding
61 for (size_t j = 0; j < embedding_dim_; j++) {
62 float val = output(i, j);
63 row_norm += val * val;
64 }
65 row_norm = std::sqrt(row_norm + eps);
66
67 // Clamp the norm to prevent division by very small values
68 row_norm = std::max(row_norm, 1e-3f);
69
70 // Scale this embedding
71 float scale = std::min(1.0f, 1.0f / row_norm);
72 for (size_t j = 0; j < embedding_dim_; j++) {
73 output(i, j) *= scale;
74 }
75 }
76
77 // Validate output
78 for (size_t i = 0; i < output.size(); i++) {
79 if (std::isnan(output.data()[i]) || std::isinf(output.data()[i])) {
80 throw std::runtime_error("Invalid value in output embeddings at position " +
81 std::to_string(i));
82 }
83 }
84
85 return output;
86}

88Matrix TokenEmbedding::project_to_vocab(const Matrix& hidden_states) {
89 Matrix logits(hidden_states.rows(), vocab_size_);
90 for (size_t i = 0; i < hidden_states.rows(); ++i) {
91 for (size_t v = 0; v < vocab_size_; ++v) {
92 float sum = 0.0f;
93 for (size_t h = 0; h < embedding_dim_; ++h) {
94 sum += hidden_states(i, h) * weights_(v, h);
95 }
96 // Prevent extreme values in logits
97 sum = std::clamp(sum, -100.0f, 100.0f);
98 logits(i, v) = sum;
99 }
100 }
101 return logits;

121 // Debug dimension information
122 std::cout << "Gradient output dimensions: " << grad_output.rows() << "x" << grad_output.cols()
123 << "\n";

125 << "\n";

129 if (grad_output.cols() != embedding_dim_) {

133 }
134
135 // Add check for input_tokens size matching grad_output rows
136 if (input_tokens.size() != grad_output.rows()) {
137 std::cout << "Warning: Input tokens size (" << input_tokens.size()
138 << ") doesn't match gradient rows (" << grad_output.rows()
139 << "). Using minimum of the two." << std::endl;
140 }
141
142 // Initialize gradient accumulator matrix with same dimensions as weights
143 Matrix weight_grads(weights_.rows(), weights_.cols(), 0.0f);
144 std::cout << "Weight grads dimensions: " << weight_grads.shape() << std::endl;
145
146 // Use the minimum of input_tokens size and grad_output rows to prevent out of bounds
147 size_t seq_length = std::min(input_tokens.size(), grad_output.rows());
148
149 // For each token in the input sequence
150 for (size_t i = 0; i < seq_length; i++) {
151 int token_id = input_tokens[i];
152 if (token_id >= static_cast<int>(weights_.rows())) {
153 throw std::runtime_error("Token ID " + std::to_string(token_id) +
154 " exceeds vocabulary size " + std::to_string(weights_.rows()));
155 }
156
157 // Accumulate gradients
158 for (size_t j = 0; j < embedding_dim_; j++) {
159 weight_grads(token_id, j) += grad_output(i, j);
160 }
161 }
162
163 // Apply gradients with learning rate
164 const float learning_rate = 0.01f;
165 std::cout << "Applying gradients with dimensions check...\n";
166 for (size_t i = 0; i < weights_.rows(); i++) {
167 for (size_t j = 0; j < weights_.cols(); j++) {
168 weights_(i, j) -= learning_rate * weight_grads(i, j);
169 if (weight_grads(i, j) != 0.0f) {

179 : encoding_matrix_(max_seq_length, hidden_size), max_seq_length_(max_seq_length),
180 hidden_size_(hidden_size) {
181 // Implement sinusoidal position embeddings

183 for (size_t i = 0; i < hidden_size; i += 2) {
