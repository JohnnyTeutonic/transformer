9__global__ void layernorm_backward_kernel(const float* grad_output, const float* input,
10 const float* gamma, float* grad_input, float* grad_gamma,
11 float* grad_beta, int hidden_size, int batch_size,
12 float eps) {
13 extern __shared__ float shared_mem[];
14 float* mean = shared_mem;
15 float* variance = &shared_mem[batch_size];
16
17 int tid = threadIdx.x;
18 int bid = blockIdx.x;
19
20 // Compute mean and variance for each sequence position
21 if (tid < hidden_size) {
22 float sum = 0.0f;
23 float sq_sum = 0.0f;
24 for (int i = 0; i < batch_size; i++) {
25 float val = input[i * hidden_size + tid];
26 sum += val;
27 sq_sum += val * val;
28 }
29 mean[tid] = sum / batch_size;
30 variance[tid] = sq_sum / batch_size - mean[tid] * mean[tid] + eps;
31 }
32 __syncthreads();
33
34 // Compute gradients
35 for (int i = tid; i < batch_size * hidden_size; i += blockDim.x) {
36 int seq_pos = i / hidden_size;
37 int hidden_pos = i % hidden_size;
38
39 float x = input[i];
40 float dy = grad_output[i];
41 float mu = mean[hidden_pos];
42 float var = variance[hidden_pos];
43 float std = sqrt(var);
44 float gamma_val = gamma[hidden_pos];
45
46 // Gradient with respect to input
47 float dx = gamma_val * (dy - mu) / std;
48 grad_input[i] = dx;
49
50 // Gradient with respect to gamma and beta
51 atomicAdd(&grad_gamma[hidden_pos], dy * (x - mu) / std);
52 atomicAdd(&grad_beta[hidden_pos], dy);
53 }
54}
