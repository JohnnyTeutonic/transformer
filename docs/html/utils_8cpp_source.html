10float Utils::adjust_learning_rate(float current_lr, float loss_ratio, size_t step) {
11 const size_t WARMUP_STEPS = 50;
12 const float PEAK_LR = 5e-4;
13 const float MIN_LR = 1e-5;
14
15 if (step < WARMUP_STEPS) {
16 return MIN_LR + (PEAK_LR - MIN_LR) * (static_cast<float>(step) / WARMUP_STEPS);
17 }
18
19 const size_t DECAY_STEPS = 5000;
20 float progress = static_cast<float>(step - WARMUP_STEPS) / DECAY_STEPS;
21 progress = std::min(1.0f, progress);
22
23 float decay_factor = 0.5f * (1.0f + std::cos(progress * M_PI));
24 float lr = MIN_LR + (PEAK_LR - MIN_LR) * decay_factor;
25
26 const float LOSS_SPIKE_THRESHOLD = 1.5f;
27 if (loss_ratio > LOSS_SPIKE_THRESHOLD) {
28 lr *= 0.1f;
29 }
30
31 return std::clamp(lr, MIN_LR, PEAK_LR);
32}

45float Utils::compute_batch_loss(const Matrix& logits, const Matrix& target_distribution) {
46 float loss = 0.0f;
47 const float epsilon = 1e-10f;
48
49 // Compute cross-entropy loss
50 for (size_t i = 0; i < logits.rows(); i++) {
51 // Find max logit for numerical stability
52 float max_logit = -std::numeric_limits<float>::infinity();
53 for (size_t j = 0; j < logits.cols(); j++) {
54 max_logit = std::max(max_logit, logits(i, j));
55 }
56
57 // Compute softmax with improved numerical stability
58 float sum_exp = 0.0f;
59 std::vector<float> probs(logits.cols());
60
61 for (size_t j = 0; j < logits.cols(); j++) {
62 probs[j] = std::exp(logits(i, j) - max_logit);
63 sum_exp += probs[j];
64 }
65
66 // Compute cross-entropy loss
67 for (size_t j = 0; j < logits.cols(); j++) {
68 probs[j] /= (sum_exp + epsilon);
69 if (target_distribution(i, j) > 0.0f) {
70 loss -= target_distribution(i, j) * std::log(probs[j] + epsilon);
71 }
72 }
73 }
74
75 // Return average loss
76 return loss / logits.rows();
77}

279 std::vector<float> last_logits;
280 for (size_t i = 0; i < logits.cols(); ++i) {
281 last_logits.push_back(logits(logits.rows() - 1, i));

289 probs[i] = std::exp(last_logits[i] - max_logit);
290 sum_exp += probs[i];

294 prob /= sum_exp;

306 [](const auto& a, const auto& b) { return a.first > b.first; });
307 }

310 for (size_t i = 0; i < std::min(k, scores.size()); ++i) {

438void Utils::apply_sampling_parameters(std::vector<float>& logits, float temperature, float top_p) {
439 // Apply temperature scaling first
440 if (temperature != 1.0f) {
441 for (auto& logit : logits) {
442 logit /= temperature;
443 }
444 }
445
446 // Apply top-p (nucleus) sampling if enabled
447 if (top_p < 1.0f) {
448 // Convert logits to probabilities
449 std::vector<std::pair<float, size_t>> probs_with_indices;
450 float max_logit = *std::max_element(logits.begin(), logits.end());
451 float sum_exp = 0.0f;
452
453 for (size_t i = 0; i < logits.size(); i++) {
454 float prob = std::exp(logits[i] - max_logit);
455 sum_exp += prob;
456 probs_with_indices.push_back({prob, i});
457 }
458
459 // Normalize probabilities
460 for (auto& pair : probs_with_indices) {
461 pair.first /= sum_exp;
462 }
463
464 // Sort by probability in descending order
465 std::sort(probs_with_indices.begin(), probs_with_indices.end(),
466 std::greater<std::pair<float, size_t>>());
467
468 // Find cutoff index for top-p
469 float cumsum = 0.0f;
470 size_t cutoff_idx = probs_with_indices.size() - 1;
471 for (size_t i = 0; i < probs_with_indices.size(); i++) {
472 cumsum += probs_with_indices[i].first;
473 if (cumsum > top_p) {
474 cutoff_idx = i;
475 break;
476 }
477 }
478
479 // Create mask for filtered tokens
480 std::vector<bool> keep_token(logits.size(), false);
481 for (size_t i = 0; i <= cutoff_idx; i++) {
482 keep_token[probs_with_indices[i].second] = true;
483 }
484
485 // Apply mask to logits
486 for (size_t i = 0; i < logits.size(); i++) {
487 if (!keep_token[i]) {
488 logits[i] = -std::numeric_limits<float>::infinity();
489 }
490 }
491 }
