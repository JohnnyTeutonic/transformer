88bool ModelSaver::saveCheckpoint(const Transformer& transformer, const std::string& directory,
89 const std::string& model_name, int epoch, float loss) {
90 try {
91 // Create directory first and check permissions
92 fs::path dir_path(directory);
93 if (!fs::exists(dir_path)) {
94 if (!fs::create_directories(dir_path)) {
95 logger.log("Failed to create directory: " + directory +
96 " (Check permissions and path)",
97 true);
98 return false;
99 }
100 } else if (!fs::is_directory(dir_path)) {
101 logger.log("Path exists but is not a directory: " + directory, true);
102 return false;
103 }
104
105 // Check directory permissions
106 std::error_code ec;
107 auto perms = fs::status(dir_path, ec).permissions();
108 if (ec) {
109 logger.log("Failed to check directory permissions: " + ec.message(), true);
110 return false;
111 }
112
113 if ((perms & fs::perms::owner_write) == fs::perms::none) {
114 logger.log("Directory is not writable: " + directory, true);
115 return false;
116 }
117
118 std::string checkpoint_file = getCheckpointFilename(directory, model_name, epoch);
119 logger.log("Saving checkpoint to: " + checkpoint_file);
120
121 // Test file writability before proceeding
122 {
123 std::ofstream test_file(checkpoint_file);
124 if (!test_file) {
125 logger.log("Cannot write to checkpoint file: " + checkpoint_file +
126 " (Check permissions)",
127 true);
128 return false;
129 }
130 }
131
132 // Open checkpoint file for actual writing
133 std::ofstream ckpt_file(checkpoint_file, std::ios::binary);
134 if (!ckpt_file) {
135 logger.log("Failed to open checkpoint file for writing: " + checkpoint_file, true);
136 return false;
137 }
138
139 // Write metadata as JSON to start of file
140 json checkpoint_meta;
141 const auto& config = transformer.getConfig();
142
143 checkpoint_meta["epoch"] = epoch;
144 checkpoint_meta["loss"] = loss;
145 checkpoint_meta["timestamp"] = std::chrono::system_clock::now().time_since_epoch().count();
146 checkpoint_meta["model_config"] = {{"vocab_size", config.vocab_size},
147 {"hidden_size", config.hidden_size},
148 {"num_heads", config.num_heads},
149 {"num_layers", config.num_layers}};
150 checkpoint_meta["batch_size"] = config.batch_size;
151
152 std::string meta_str = checkpoint_meta.dump();
153 size_t meta_size = meta_str.size();
154
155 // Write metadata size and content
156 if (!ckpt_file.write(reinterpret_cast<const char*>(&meta_size), sizeof(meta_size)) ||
157 !ckpt_file.write(meta_str.c_str(), meta_size)) {
158 logger.log("Failed to write metadata to checkpoint file", true);
159 return false;
160 }
161
162 // Save model state
163 const auto& layers = transformer.getLayers();
164 for (const auto& layer : layers) {
165 layer->save(ckpt_file);
166 }
167
168 // Ensure everything is written
169 ckpt_file.flush();
170 if (!ckpt_file) {
171 logger.log("Error occurred while writing checkpoint file", true);
172 return false;
173 }
174
175 logger.log("Checkpoint saved successfully");
176 return true;
177 } catch (const std::exception& e) {
178 logger.log("Error saving checkpoint: " + std::string(e.what()), true);
179 return false;
180 }

183bool ModelSaver::loadCheckpoint(Transformer& transformer, const std::string& checkpoint_path) {
184 std::ifstream ckpt_file(checkpoint_path, std::ios::binary);
185 if (!ckpt_file) {
186 logger.log("Failed to open checkpoint file for reading", true);
187 return false;
188 }
189
190 try {
191 // Read metadata size
192 size_t meta_size;
193 ckpt_file.read(reinterpret_cast<char*>(&meta_size), sizeof(meta_size));
194
195 // Read metadata JSON
196 std::string meta_str(meta_size, '\0');
197 ckpt_file.read(&meta_str[0], meta_size);
198
199 json checkpoint_meta = json::parse(meta_str);
200
201 // Verify model configuration
202 const auto& config = transformer.getConfig();
203 const auto& saved_config = checkpoint_meta["model_config"];
204
205 if (saved_config["vocab_size"] != config.vocab_size ||
206 saved_config["hidden_size"] != config.hidden_size ||
207 saved_config["num_heads"] != config.num_heads ||
208 saved_config["num_layers"] != config.num_layers) {
209 logger.log("Model configuration mismatch in checkpoint", true);
210 return false;
211 }
212
213 // Load model state
214 transformer.load(ckpt_file);
215
216 logger.log("Successfully loaded checkpoint from epoch " +
217 std::to_string(checkpoint_meta["epoch"].get<int>()));
218 return true;
219 } catch (const std::exception& e) {
220 logger.log("Error loading checkpoint: " + std::string(e.what()), true);
221 return false;

225bool ModelSaver::loadLatestCheckpoint(Transformer& transformer, const std::string& directory,
226 const std::string& model_name, int& epoch, float& loss) {
227 try {
228 // Find latest checkpoint
229 int latest_epoch = -1;
230 for (const auto& entry : fs::directory_iterator(directory)) {
231 if (entry.path().extension() == ".ckpt") {
232 std::string filename = entry.path().stem().string();
233 if (filename.find(model_name + "_checkpoint_") == 0) {
234 int checkpoint_epoch =
235 std::stoi(filename.substr(filename.find_last_of("_") + 1));
236 latest_epoch = std::max(latest_epoch, checkpoint_epoch);
237 }
238 }
239 }
240
241 if (latest_epoch == -1) {
242 logger.log("No checkpoints found", true);
243 return false;
244 }
245
246 // Load the latest checkpoint
247 std::string checkpoint_file = getCheckpointFilename(directory, model_name, latest_epoch);
248 if (!loadCheckpoint(transformer, checkpoint_file)) {
249 return false;
250 }
251
252 // Update epoch and loss from the loaded checkpoint
253 std::ifstream ckpt_file(checkpoint_file, std::ios::binary);
254 size_t meta_size;
255 ckpt_file.read(reinterpret_cast<char*>(&meta_size), sizeof(meta_size));
256
257 std::string meta_str(meta_size, '\0');
258 ckpt_file.read(&meta_str[0], meta_size);
259
260 json checkpoint_meta = json::parse(meta_str);
261 epoch = checkpoint_meta["epoch"];
262 loss = checkpoint_meta["loss"];
263
264 return true;
265 } catch (const std::exception& e) {
266 logger.log("Error loading latest checkpoint: " + std::string(e.what()), true);
267 return false;
