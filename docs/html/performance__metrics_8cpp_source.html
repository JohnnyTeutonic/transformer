43void PerformanceMetrics::record_attention_flops(size_t seq_length, size_t num_heads,
44 size_t head_dim) {
45 // Calculate FLOPs for attention computation
46 // Q*K^T: 2*seq_length^2*head_dim operations per head
47 // Softmax: 3*seq_length operations per head
48 // Attention*V: 2*seq_length^2*head_dim operations per head
49 size_t flops = num_heads * (4 * seq_length * seq_length * head_dim + 3 * seq_length);
50 accumulated_times["attention_flops"] += flops;
51 call_counts["attention_flops"]++;
52}
