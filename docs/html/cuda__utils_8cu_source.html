20__global__ void softmax_kernel(float* scores, int seq_len) {
21 int row = blockIdx.x * blockDim.x + threadIdx.x;
22
23 if (row < seq_len) {
24 float max_val = scores[row * seq_len];
25 for (int i = 1; i < seq_len; i++) {
26 max_val = max(max_val, scores[row * seq_len + i]);
27 }
28
29 float sum = 0.0f;
30 for (int i = 0; i < seq_len; i++) {
31 scores[row * seq_len + i] = expf(scores[row * seq_len + i] - max_val);
32 sum += scores[row * seq_len + i];
33 }
34

36 scores[row * seq_len + i] /= sum;
