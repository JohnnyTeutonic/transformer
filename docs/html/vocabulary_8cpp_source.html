367bool Vocabulary::verify_mappings() const {
368 bool valid = true;
369 for (const auto& [token, id] : token_to_id) {
370 if (id >= id_to_token.size()) {
371 std::cout << "Error: Token '" << token << "' maps to invalid ID " << id << "\n";
372 valid = false;
373 continue;
374 }
375 if (id_to_token[id] != token) {
