46 // Scale factor for attention scores
47 float scale = 1.0f / std::sqrt(static_cast<float>(head_dim));
48
49 // Compute attention scores
50 Matrix scores = matmul(Q, K.transpose());
51 scores *= scale;
52
53 // Apply mask if provided
54 if (!mask.mask.empty()) {
55 for (size_t i = 0; i < scores.rows(); i++) {
56 for (size_t j = 0; j < scores.cols(); j++) {
57 if (mask.mask(i % mask.mask.rows(), j % mask.mask.cols()) == 0.0f) {
58 scores(i, j) = -std::numeric_limits<float>::infinity();
59 }
60 }
61 }
62 }
63
64 // Apply softmax
65 for (size_t i = 0; i < scores.rows(); i++) {
66 float max_val = -std::numeric_limits<float>::infinity();
67 for (size_t j = 0; j < scores.cols(); j++) {
68 max_val = std::max(max_val, scores(i, j));
69 }
70
71 float sum = 0.0f;
72 for (size_t j = 0; j < scores.cols(); j++) {
73 scores(i, j) = std::exp(scores(i, j) - max_val);
74 sum += scores(i, j);
75 }
76
77 for (size_t j = 0; j < scores.cols(); j++) {
78 scores(i, j) /= sum;
79 }
80 }
