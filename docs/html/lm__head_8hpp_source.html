90 // Compute gradients for projection and bias
91 std::cout << "Computing gradients for projection and bias" << std::endl;
92 Matrix grad_proj = matmul(grad_output.transpose(), hidden_states);
93 std::cout << "grad projection shape: " << grad_proj.shape() << std::endl;
94 Vector grad_bias = grad_output.row_sum();
95 std::cout << "grad bias size: " << grad_bias.size() << std::endl;
96
97 // Apply weight updates with adaptive learning rate
98 float lr = 0.001f; // Base learning rate
99 float beta1 = 0.9f; // Momentum parameter
100 float beta2 = 0.999f; // RMSprop parameter
101 float eps = 1e-8f; // Small constant for numerical stability
102
103 static Matrix m_proj(projection.rows(), projection.cols(),
104 0.0f); // Momentum for projection
105 static Matrix v_proj(projection.rows(), projection.cols(),
106 0.0f); // RMSprop for projection
107 static Vector m_bias(bias.size(), 0.0f); // Momentum for bias
108 static Vector v_bias(bias.size(), 0.0f); // RMSprop for bias
109 static size_t t = 0; // Time step
110 t++;
111
112 // Update projection matrix using Adam optimizer
113 std::cout << "updating projection matrix using Adam optimizer" << std::endl;
114 for (size_t i = 0; i < projection.rows(); ++i) {
115 for (size_t j = 0; j < projection.cols(); ++j) {
116 std::cout << "updating momentum" << std::endl;
117 // Update momentum
118 m_proj(i, j) = beta1 * m_proj(i, j) + (1 - beta1) * grad_proj(i, j);
119 std::cout << "updating RMSprop" << std::endl;
120 // Update RMSprop
121 v_proj(i, j) =
122 beta2 * v_proj(i, j) + (1 - beta2) * grad_proj(i, j) * grad_proj(i, j);
123 std::cout << "calculating bias correction" << std::endl;
124 // Bias correction
125 float m_hat = m_proj(i, j) / (1 - std::pow(beta1, t));
126 float v_hat = v_proj(i, j) / (1 - std::pow(beta2, t));
127 std::cout << "updating weights" << std::endl;
128 // Update weights
129 projection(i, j) -= lr * m_hat / (std::sqrt(v_hat) + eps);
130 }
131 }
132
133 // Update bias vector using Adam optimizer
134 for (size_t i = 0; i < bias.size(); ++i) {
135 std::cout << "updating momentum" << std::endl;
136 // Update momentum
137 m_bias[i] = beta1 * m_bias[i] + (1 - beta1) * grad_bias[i];
138 std::cout << "updating RMSprop" << std::endl;
139 // Update RMSprop
140 v_bias[i] = beta2 * v_bias[i] + (1 - beta2) * grad_bias[i] * grad_bias[i];
141 std::cout << "calculating bias correction" << std::endl;
142 // Bias correction
143 float m_hat = m_bias[i] / (1 - std::pow(beta1, t));
144 float v_hat = v_bias[i] / (1 - std::pow(beta2, t));
145 std::cout << "updating bias" << std::endl;
146 // Update bias
147 bias[i] -= lr * m_hat / (std::sqrt(v_hat) + eps);
148 }
149 std::cout << "Gradient with respect to input" << std::endl;
150 std::cout << "grad_output dims: " << grad_output.rows() << "x" << grad_output.cols()
151 << std::endl;
152 std::cout << "projection dims: " << projection.rows() << "x" << projection.cols()
153 << std::endl;
154 // Compute gradient with respect to input
155 Matrix grad_input = matmul(grad_output, projection);
156 if (grad_input.cols() != hidden_states.cols()) {
157 throw std::runtime_error("Language model head gradient output dimension (" +
158 std::to_string(grad_input.cols()) +

161 }
