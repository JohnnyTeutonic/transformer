36Matrix TransformerLayer::forward(const Matrix& input, const AttentionMask& mask,
37 const std::optional<KVCache>& kv_cache) {
38 std::cout << "=== TransformerLayer::forward START ===" << std::endl;
39
40 // Layer norm before attention
41 Matrix normalized = attention_ln->forward(input);
42
43 // Cache the normalized input for attention backward pass
44 std::string attn_key = "attn_norm_" + std::to_string(layer_idx);
45 GradientCheckpoint::cache_activation(attn_key, normalized);
46
47 // Self attention
48 Matrix attention_output = self_attention->forward(normalized, mask, kv_cache);
49 if (training) {
50 attention_output = attention_dropout->forward(attention_output, true);
51 }

53 std::cout << "calculating attention ln" << std::endl;
54 Matrix norm1 = attention_ln->forward(residual);
55
56 // Cache the normalized input for feed forward backward pass
57 std::string ffn_key = "ffn_norm_" + std::to_string(layer_idx);

62 if (training) {

194Matrix Transformer::forward(const std::vector<int>& input_tokens, bool use_cache) {
195 static const bool use_fp16 = config.use_fp16;
196
197 // Get embeddings
198 Matrix embeddings = token_embedding->forward(input_tokens);
199
200 if (use_fp16) {
201 HalfPrecisionTraining::convert_to_fp16(embeddings);
202 }
203
204 // Add positional encodings
205 Matrix position_ids(input_tokens.size(), 1);
206 for (size_t i = 0; i < input_tokens.size(); ++i) {
207 position_ids(i, 0) = static_cast<float>(i);
208 }
209 Matrix pos_encodings = pos_encoding->forward(position_ids);
210
211 if (use_fp16) {
212 HalfPrecisionTraining::convert_to_fp16(pos_encodings);
213 }
214
215 embeddings += pos_encodings;
216
217 // Create causal mask for next-token prediction
218 AttentionMask mask = AttentionMask::create_causal_mask(input_tokens.size());
219
220 // Forward through layers with stability checks
221 hidden_states = embeddings;
222 m_layer_activations.clear(); // Clear previous activations
223 m_layer_activations.reserve(layers.size()); // Reserve space for efficiency
224
225 // Add dropout after embeddings
226 if (training && dropout) {
227 hidden_states = dropout->forward(hidden_states, true);
228 }
229
230 for (size_t i = 0; i < layers.size(); ++i) {
231 try {
232 m_layer_activations.push_back(hidden_states);
233 hidden_states = layers[i]->forward(hidden_states, mask,
234 use_cache ? std::optional<KVCache>(m_kv_caches[i])
235 : std::nullopt);
236
237 // Convert back to FP32 at the end
238 if (use_fp16 && i == layers.size() - 1) {
239 HalfPrecisionTraining::convert_to_fp32(hidden_states);
240 }
241
242 // Add dropout between layers
243 if (training && dropout && i < layers.size() - 1) {
244 hidden_states = dropout->forward(hidden_states, true);
245 }
246 } catch (const std::exception& e) {
247 std::cerr << "Error in layer " << i << ": " << e.what() << std::endl;
248 throw;
249 }
250 }
251
252 // Store final hidden states for backward pass
253 last_hidden_states = hidden_states;
254
255 return hidden_states;

264void Transformer::backward(const Matrix& grad_output, const std::vector<int>& input_tokens,
265 float learning_rate) {
266 static const bool use_fp16 = config.use_fp16;
267
268 Matrix grad = grad_output;
269 if (use_fp16) {
270 HalfPrecisionTraining::convert_to_fp16(grad);
271 }
272
273 const float grad_clip_threshold = 1.0f;
274
275 // Clip incoming gradients
276 Matrix clipped_grad = grad;
277 float grad_norm = 0.0f;
278
279 // Compute gradient norm
280 for (size_t i = 0; i < clipped_grad.size(); ++i) {
281 grad_norm += clipped_grad.data()[i] * clipped_grad.data()[i];
282 }
283 grad_norm = std::sqrt(grad_norm);
284
285 // Apply clipping if norm is too large
286 if (grad_norm > grad_clip_threshold) {
287 float scale = grad_clip_threshold / (grad_norm + 1e-6f);
288 for (size_t i = 0; i < clipped_grad.size(); ++i) {
289 clipped_grad.data()[i] *= scale;
290 }
291 }
292
293 // Continue with backward pass using clipped gradients
294 Matrix current_grad = clipped_grad;
295 std::cout << "\n=== Transformer::backward START ===" << std::endl;
296
297 std::cout << "Initial grad dimensions: " << current_grad.rows() << "x" << current_grad.cols()
298 << std::endl;
299
300 // Backward through final layer norm
301 const Matrix& last_activation = m_layer_activations.back();
302 std::cout << "Last activation dimensions: " << last_activation.rows() << "x"
303 << last_activation.cols() << std::endl;
304
305 if (current_grad.rows() != last_activation.rows() ||
306 current_grad.cols() != last_activation.cols()) {
307 throw std::runtime_error("Dimension mismatch in final layer norm backward: grad(" +
308 std::to_string(current_grad.rows()) + "," +
309 std::to_string(current_grad.cols()) + ") != activation(" +
310 std::to_string(last_activation.rows()) + "," +
311 std::to_string(last_activation.cols()) + ")");
312 }
313
314 current_grad = final_ln->backward(current_grad, last_activation);
315 std::cout << "After final LN grad dimensions: " << current_grad.rows() << "x"
316 << current_grad.cols() << std::endl;
317
318 // Backward through layers in reverse order
319 for (int i = layers.size() - 1; i >= 0; --i) {
320 std::cout << "\nProcessing layer " << i << std::endl;
321 const Matrix& layer_input = m_layer_activations[i];
322 std::cout << "Layer input dimensions: " << layer_input.rows() << "x" << layer_input.cols()
323 << std::endl;
324 std::cout << "Current grad dimensions: " << current_grad.rows() << "x"
325 << current_grad.cols() << std::endl;
326
327 if (current_grad.rows() != layer_input.rows() ||
328 current_grad.cols() != layer_input.cols()) {
329 throw std::runtime_error("Dimension mismatch in layer " + std::to_string(i) +
330 " backward: grad(" + std::to_string(current_grad.rows()) +
331 "," + std::to_string(current_grad.cols()) + ") != input(" +
332 std::to_string(layer_input.rows()) + "," +
333 std::to_string(layer_input.cols()) + ")");
334 }
335
336 current_grad = layers[i]->backward(current_grad, layer_input, Matrix());
337 std::cout << "After layer " << i << " grad dimensions: " << current_grad.rows() << "x"
338 << current_grad.cols() << std::endl;
339 }
340
341 // Convert back to FP32 before parameter updates
342 if (use_fp16) {
343 HalfPrecisionTraining::convert_to_fp32(current_grad);
344 }
345
346 // Update parameters with L2 regularization
347 for (auto& param : parameters()) {
348 // L2 regularization gradient
349 Matrix l2_grad = param;
350 l2_grad *= config.weight_decay;
351
352 // Get corresponding gradient
353 auto& param_grads = parameter_gradients();
354 size_t param_idx = &param - &parameters()[0];
355 Matrix& param_grad = param_grads[param_idx];
356
357 // Combine with existing gradients
358 param_grad += l2_grad;
359
360 // Update parameters
361 param -= param_grad * learning_rate;
362 }
363
364 std::cout << "=== Transformer::backward END ===\n" << std::endl;
365}
